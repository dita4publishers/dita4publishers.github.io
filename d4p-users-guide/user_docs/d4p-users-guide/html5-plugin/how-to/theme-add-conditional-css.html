      <name>style</name>
      <output>yes</output>
      <prefix><![CDATA[<!--[if IE]>]]></prefix>
      <suffix><![CDATA[<![endif]-->]]></suffix>
      <value>

        .my-class {
          /** ... **/
        }

      </value>
    </tag>
